  140.   public function fb($Object) {
    141. 

  141.   
    142. 

  142.     if (headers_sent($filename, $linenum)) {
    143. 

  143.         throw $this->newException('Headers already sent in '.$filename.' on line '.$linenum.'. Cannot send log data to FirePHP. You must have Output Buffering enabled via ob_start() or output_buffering ini directive.');
    144. 

  144.     }
    145. 

  145.   
    146. 

  146.     $Type = null;
    147. 

  147.   
    148. 

  148.     if(func_num_args()==1) {
    149. 

  149.     } else
    150. 

  150.     if(func_num_args()==2) {
    151. 

  151.       switch(func_get_arg(1)) {
    152. 

  152.         case self::LOG:
    153. 

  153.         case self::INFO:
    154. 

  154.         case self::WARN:
    155. 

  155.         case self::ERROR:
    156. 

  156.         case self::DUMP:
    157. 

  157.         case self::EXCEPTION:
    158. 

  158.         case self::TABLE:
    159. 

  159.           $Type = func_get_arg(1);
    160. 

  160.           break;
    161. 

  161.         default:
    162. 

  162.           $Object = array(func_get_arg(1),$Object);
    163. 

  163.           break;
    164. 

  164.       }
    165. 

  165.     } else
    166. 

  166.     if(func_num_args()==3) {
    167. 

  167.       $Type = func_get_arg(2);
    168. 

  168.       $Object = array(func_get_arg(1),$Object);
    169. 

  169.     } else {
    170. 

  170.       throw $this->newException('Wrong number of arguments to fb() function!');
    171. 

  171.     }
    172. 

  172.   
    173. 

  173.   
    174. 

  174.     if(!$this->detectClientExtension()) {
    175. 

  175.       return false;
    176. 

  176.     }
    177. 

  177.   
    178. 

  178.     if($Object instanceof Exception) {
    179. 

  179.       
    180. 

  180.       $Object = array('Class'=>get_class($Object),
    181. 

  181.                       'Message'=>$Object->getMessage(),
    182. 

  182.                       'File'=>$Object->getFile(),
    183. 

  183.                       'Line'=>$Object->getLine(),
    184. 

  184.                       'Trace'=>$Object->getTrace());
    185. 

  185.       if($Type===null || $Type===self::EXCEPTION) {
    186. 

  186.         $Type = 'TRACE';
    187. 

  187.       }
    188. 

  188.       
    189. 

  189.     } else {
    190. 

  190.       if($Type===null) {
    191. 

  191.         $Type = self::LOG;
    192. 

  192.       }
    193. 

  193.     }
    194. 

  194.   
    195. 

  195.       $this->setHeader('X-FirePHP-Data-100000000001','{');
    196. 

  196.     if($Type==self::DUMP) {
    197. 

  197.         $this->setHeader('X-FirePHP-Data-200000000001','"FirePHP.Dump":{');
    198. 

  198.         $this->setHeader('X-FirePHP-Data-299999999999','"__SKIP__":"__SKIP__"},');
    199. 

  199.     } else {
    200. 

  200.         $this->setHeader('X-FirePHP-Data-300000000001','"FirePHP.Firebug.Console":[');
    201. 

  201.         $this->setHeader('X-FirePHP-Data-399999999999','["__SKIP__"]],');
    202. 

  202.     }
    203. 

  203.       $this->setHeader('X-FirePHP-Data-999999999999','"__SKIP__":"__SKIP__"}');
    204. 

  204.   
    205. 

  205.     if($Type==self::DUMP) {
    206. 

  206.         $msg = '"'.$Object[0].'":'.$this->json_encode($Object[1]).',';
    207. 

  207.     } else {
    208. 

  208.         $msg = '["'.$Type.'",'.$this->json_encode($Object).'],';
    209. 

  209.     }
    210. 

  210.    
    211. 

  211.       foreach( explode("\n",chunk_split($msg, 5000, "\n")) as $part ) {
    212. 

  212.         
    213. 

  213.       if($part) {
    214. 

  214.  
    215. 

  215.         usleep(1); /* Ensure microtime() increments with each loop. Not very elegant but it works */
    216. 

  216.     
    217. 

  217.             $mt = explode(' ',microtime());
    218. 

  218.             $mt = substr($mt[1],7).substr($mt[0],2);
    219. 

  219.     
    220. 

  220.         $this->setHeader('X-FirePHP-Data-'.(($Type==self::DUMP)?'2':'3').$mt, $part);
    221. 

  221.       }
    222. 

  222.       }
    223. 

  223.     
    224. 

  224.     return true;
    225. 

  225.   }

  303.  /**
    304. 

  304.   * convert a string from one UTF-8 char to one UTF-16 char
    305. 

  305.   *
    306. 

  306.   * Normally should be handled by mb_convert_encoding, but
    307. 

  307.   * provides a slower PHP-only method for installations
    308. 

  308.   * that lack the multibye string extension.
    309. 

  309.   *
    310. 

  310.   * @param    string  $utf8   UTF-8 character
    311. 

  311.   * @return   string  UTF-16 character
    312. 

  312.   * @access   private
    313. 

  313.   */
    314. 

  314.   private function json_utf82utf16($utf8)
    315. 

  315.   {
    316. 

  316.       // oh please oh please oh please oh please oh please
    317. 

  317.       if(function_exists('mb_convert_encoding')) {
    318. 

  318.           return mb_convert_encoding($utf8, 'UTF-16', 'UTF-8');
    319. 

  319.       }
    320. 

  320.  
    321. 

  321.       switch(strlen($utf8)) {
    322. 

  322.           case 1:
    323. 

  323.               // this case should never be reached, because we are in ASCII range
    324. 

  324.               // see: http://www.cl.cam.ac.uk/~mgk25/unicode.html#utf-8
    325. 

  325.               return $utf8;
    326. 

  326.  
    327. 

  327.           case 2:
    328. 

  328.               // return a UTF-16 character from a 2-byte UTF-8 char
    329. 

  329.               // see: http://www.cl.cam.ac.uk/~mgk25/unicode.html#utf-8
    330. 

  330.               return chr(0x07 & (ord($utf8{0}) >> 2))
    331. 

  331.                    . chr((0xC0 & (ord($utf8{0}) << 6))
    332. 

  332.                        | (0x3F & ord($utf8{1})));
    333. 

  333.  
    334. 

  334.           case 3:
    335. 

  335.               // return a UTF-16 character from a 3-byte UTF-8 char
    336. 

  336.               // see: http://www.cl.cam.ac.uk/~mgk25/unicode.html#utf-8
    337. 

  337.               return chr((0xF0 & (ord($utf8{0}) << 4))
    338. 

  338.                        | (0x0F & (ord($utf8{1}) >> 2)))
    339. 

  339.                    . chr((0xC0 & (ord($utf8{1}) << 6))
    340. 

  340.                        | (0x7F & ord($utf8{2})));
    341. 

  341.       }
    342. 

  342.  
    343. 

  343.       // ignoring UTF-32 for now, sorry
    344. 

  344.       return '';
    345. 

  345.   }

  347.  /**
    348. 

  348.   * encodes an arbitrary variable into JSON format
    349. 

  349.   *
    350. 

  350.   * @param    mixed   $var    any number, boolean, string, array, or object to be encoded.
    351. 

  351.   *                            see argument 1 to Services_JSON() above for array-parsing behavior.
    352. 

  352.   *                            if var is a strng, note that encode() always expects it
    353. 

  353.   *                            to be in ASCII or UTF-8 format!
    354. 

  354.   *
    355. 

  355.   * @return   mixed   JSON string representation of input var or an error if a problem occurs
    356. 

  356.   * @access   public
    357. 

  357.   */
    358. 

  358.   private function json_encode($var)
    359. 

  359.   {
    360. 

  360.     
    361. 

  361.     if(is_object($var)) {
    362. 

  362.       if(in_array($var,$this->json_objectStack)) {
    363. 

  363.         return '"** Recursion **"';
    364. 

  364.       }
    365. 

  365.     }
    366. 

  366.           
    367. 

  367.       switch (gettype($var)) {
    368. 

  368.           case 'boolean':
    369. 

  369.               return $var ? 'true' : 'false';
    370. 

  370.  
    371. 

  371.           case 'NULL':
    372. 

  372.               return 'null';
    373. 

  373.  
    374. 

  374.           case 'integer':
    375. 

  375.               return (int) $var;
    376. 

  376.  
    377. 

  377.           case 'double':
    378. 

  378.           case 'float':
    379. 

  379.               return (float) $var;
    380. 

  380.  
    381. 

  381.           case 'string':
    382. 

  382.               // STRINGS ARE EXPECTED TO BE IN ASCII OR UTF-8 FORMAT
    383. 

  383.               $ascii = '';
    384. 

  384.               $strlen_var = strlen($var);
    385. 

  385.  
    386. 

  386.              /*
    387. 

  387.               * Iterate over every character in the string,
    388. 

  388.               * escaping with a slash or encoding to UTF-8 where necessary
    389. 

  389.               */
    390. 

  390.               for ($c = 0; $c < $strlen_var; ++$c) {
    391. 

  391.  
    392. 

  392.                   $ord_var_c = ord($var{$c});
    393. 

  393.  
    394. 

  394.                   switch (true) {
    395. 

  395.                       case $ord_var_c == 0x08:
    396. 

  396.                           $ascii .= '\b';
    397. 

  397.                           break;
    398. 

  398.                       case $ord_var_c == 0x09:
    399. 

  399.                           $ascii .= '\t';
    400. 

  400.                           break;
    401. 

  401.                       case $ord_var_c == 0x0A:
    402. 

  402.                           $ascii .= '\n';
    403. 

  403.                           break;
    404. 

  404.                       case $ord_var_c == 0x0C:
    405. 

  405.                           $ascii .= '\f';
    406. 

  406.                           break;
    407. 

  407.                       case $ord_var_c == 0x0D:
    408. 

  408.                           $ascii .= '\r';
    409. 

  409.                           break;
    410. 

  410.  
    411. 

  411.                       case $ord_var_c == 0x22:
    412. 

  412.                       case $ord_var_c == 0x2F:
    413. 

  413.                       case $ord_var_c == 0x5C:
    414. 

  414.                           // double quote, slash, slosh
    415. 

  415.                           $ascii .= '\\'.$var{$c};
    416. 

  416.                           break;
    417. 

  417.  
    418. 

  418.                       case (($ord_var_c >= 0x20) && ($ord_var_c <= 0x7F)):
    419. 

  419.                           // characters U-00000000 - U-0000007F (same as ASCII)
    420. 

  420.                           $ascii .= $var{$c};
    421. 

  421.                           break;
    422. 

  422.  
    423. 

  423.                       case (($ord_var_c & 0xE0) == 0xC0):
    424.
