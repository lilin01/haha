  110.      * Retrieves config info based on the file, section, and variable name.

    112. 

  112.      * @param string $file_name config file to get info for
    113.

  117.     function get($file_name, $section_name = NULL, $var_name = NULL)
    118. 

  118.     {
    119. 

  119.         if (empty($file_name)) {
    120. 

  120.             $this->_trigger_error_msg('Empty config file name');
    121. 

  121.             return;
    122. 

  122.         } else {
    123. 

  123.             $file_name = $this->_config_path . $file_name;
    124. 

  124.             if (!isset($this->_config_data[$file_name]))
    125. 

  125.                 $this->load_file($file_name, false);
    126. 

  126.         }
    127. 

  127.  
    128. 

  128.         if (!empty($var_name)) {
    129. 

  129.             if (empty($section_name)) {
    130. 

  130.                 return $this->_config_data[$file_name]["vars"][$var_name];
    131. 

  131.             } else {
    132. 

  132.                 if(isset($this->_config_data[$file_name]["sections"][$section_name]["vars"][$var_name]))
    133. 

  133.                     return $this->_config_data[$file_name]["sections"][$section_name]["vars"][$var_name];
    134. 

  134.                 else
    135. 

  135.                     return array();
    136. 

  136.             }
    137. 

  137.         } else {
    138. 

  138.             if (empty($section_name)) {
    139. 

  139.                 return (array)$this->_config_data[$file_name]["vars"];
    140. 

  140.             } else {
    141. 

  141.                 if(isset($this->_config_data[$file_name]["sections"][$section_name]["vars"]))
    142. 

  142.                     return (array)$this->_config_data[$file_name]["sections"][$section_name]["vars"];
    143. 

  143.                 else
    144. 

  144.                     return array();
    145. 

  145.             }
    146. 

  146.         }
    147. 

  147.     }

  238.     function load_file($file_name, $prepend_path = true)
    239. 

  239.     {
    240. 

  240.         if ($prepend_path && $this->_config_path != "")
    241. 

  241.             $config_file = $this->_config_path . $file_name;
    242. 

  242.         else
    243. 

  243.             $config_file = $file_name;
    244. 

  244.  
    245. 

  245.         ini_set('track_errors', true);
    246. 

  246.         $fp = @fopen($config_file, "r");
    247. 

  247.         if (!is_resource($fp)) {
    248. 

  248.             $this->_trigger_error_msg("Could not open config file '$config_file'");
    249. 

  249.             return false;
    250. 

  250.         }
    251. 

  251.  
    252. 

  252.         $contents = ($size = filesize($config_file)) ? fread($fp, $size) : '';
    253. 

  253.         fclose($fp);
    254. 

  254.  
    255. 

  255.         $this->_config_data[$config_file] = $this->parse_contents($contents);
    256. 

  256.         return true;
    257.

  271.     /**
    272. 

  272.      * parse the source of a configuration file manually.
    273. 

  273.      *
    274. 

  274.      * @param string $contents the file-contents to parse
    275. 

  275.      */
    276. 

  276.     function parse_contents($contents)
    277. 

  277.     {
    278. 

  278.         if($this->fix_newlines) {
    279. 

  279.             // fix mac/dos formatted newlines
    280. 

  280.             $contents = preg_replace('!\r\n?!', "\n", $contents);
    281. 

  281.         }
    282. 

  282.  
    283. 

  283.         $config_data = array();
    284. 

  284.         $config_data['sections'] = array();
    285. 

  285.         $config_data['vars'] = array();
    286. 

  286.  
    287. 

  287.         /* reference to fill with data */
    288. 

  288.         $vars =& $config_data['vars'];
    289. 

  289.  
    290. 

  290.         /* parse file line by line */
    291. 

  291.         preg_match_all('!^.*\r?\n?!m', $contents, $match);
    292. 

  292.         $lines = $match[0];
    293. 

  293.         for ($i=0, $count=count($lines); $i<$count; $i++) {
    294. 

  294.             $line = $lines[$i];
    295. 

  295.             if (empty($line)) continue;
    296. 

  296.  
    297. 

  297.             if ( $line{0} == '[' && preg_match('!^\[(.*?)\]!', $line, $match) ) {
    298. 

  298.                 /* section found */
    299. 

  299.                 if ($match[1]{0} == '.') {
    300. 

  300.                     /* hidden section */
    301. 

  301.                     if ($this->read_hidden) {
    302. 

  302.                         $section_name = substr($match[1], 1);
    303. 

  303.                     } else {
    304. 

  304.                         /* break reference to $vars to ignore hidden section */
    305. 

  305.                         unset($vars);
    306. 

  306.                         $vars = array();
    307. 

  307.                         continue;
    308. 

  308.                     }
    309. 

  309.                 } else {                    
    310. 

  310.                     $section_name = $match[1];
    311. 

  311.                 }
    312. 

  312.                 if (!isset($config_data['sections'][$section_name]))
    313. 

  313.                     $config_data['sections'][$section_name] = array('vars' => array());
    314. 

  314.                 $vars =& $config_data['sections'][$section_name]['vars'];
    315. 

  315.                 continue;
    316. 

  316.             }
    317. 

  317.  
    318. 

  318.             if (preg_match('/^\s*(\.?\w+)\s*=\s*(.*)/s', $line, $match)) {
    319. 

  319.                 /* variable found */
    320. 

  320.                 $var_name = rtrim($match[1]);
    321. 

  321.                 if (strpos($match[2], '"""') === 0) {
    322. 

  322.                     /* handle multiline-value */
    323. 

  323.                     $lines[$i] = substr($match[2], 3);
    324. 

  324.                     $var_value = '';
    325. 

  325.                     while ($i<$count) {
    326. 

  326.                         if (($pos = strpos($lines[$i], '"""')) === false) {
    327. 

  327.                             $var_value .= $lines[$i++];
    328. 

  328.                         } else {
    329. 

  329.                             /* end of multiline-value */
    330. 

  330.                             $var_value .= substr($lines[$i], 0, $pos);
    331. 

  331.                             break;
    332. 

  332.                         }
    333. 

  333.                     }
    334. 

  334.                     $booleanize = false;
    335. 

  335.  
    336. 

  336.                 } else {
    337. 

  337.                     /* handle simple value */
    338. 

  338.                     $var_value = preg_replace('/^([\'"])(.*)\1$/', '\2', rtrim($match[2]));
    339. 

  339.                     $booleanize = $this->booleanize;
    340. 

  340.  
    341. 

  341.                 }
    342. 

  342.                 $this->_set_config_var($vars, $var_name, $var_value, $booleanize);
    343. 

  343.             }
    344. 

  344.             /* else unparsable line / means it is a comment / means ignore it */
    345. 

  345.         }
    346. 

  346.         return $config_data;
    347. 

  347.     }

    352. 

  352.      * @param array &$container 
    353. 

  353.      * @param string $var_name 
    354. 

  354.      * @param mixed $var_value 
    355. 

  355.      * @param boolean $booleanize determines whether $var_value is converted to
    356. 

  356.      *                             to true/false
    357. 

  357.      */
    358. 

  358.     function _set_config_var(&$container, $var_name, $var_value, $booleanize)
    359. 

  359.     {
    360. 

  360.         if ($var_name{0} == '.') {
    361. 

  361.             if (!$this->read_hidden)
    362. 

  362.                 return;
    363. 

  363.             else
    364. 

  364.                 $var_name = substr($var_name, 1);
    365. 

  365.         }
    366. 

  366.  
    367. 

  367.         if (!preg_match("/^[a-zA-Z_]\w*$/", $var_name)) {
    368. 

  368.             $this->_trigger_error_msg("Bad variable name '$var_name'");
    369. 

  369.             return;
    370. 

  370.         }
    371. 

  371.  
    372. 

  372.         if ($booleanize) {
    373. 

  373.             if (preg_match("/^(on|true|yes)$/i", $var_value))
    374. 

  374.                 $var_value = true;
    375. 

  375.             else if (preg_match("/^(off|false|no)$/i", $var_value))
    376. 

  376.                 $var_value = false;
    377. 

  377.         }
    378. 

  378.  
    379. 

  379.         if (!isset($container[$var_name]) || $this->overwrite)
    380. 

  380.             $container[$var_name] = $var_value;
    381. 

  381.         else {
    382. 

  382.             settype($container[$var_name], 'array');
    383. 

  383.             $container[$var_name][] = $var_value;
    384. 

  384.         }
    385. 

  385.     }
