    239.

  242.                 case _ORM_DT_DATETIME: $result = $this->quote(
    243.

    244. 

  244.                 case _ORM_DT_DATE: $result = $this->quote(
    245.

  246.                 case _ORM_DT_TIME: $result = $this->quote(
    247. 

  247.                         strftime('%H:%M:%S', is_numeric($value) ? $value : strtotime($value))); break;
    248. 

  248.                 case _ORM_DT_SQL: $result = $this->encodeFieldName($value); break;
    249. 

  249.                 default: $result = $this->quote($value); break;
    250. 

  250.             }

  251.             return $result;

  359.                         break;

  360.                     case _ORM_OP_IN:

  361.                         if (is_array($value)) {
    362. 

  362.                             $sql .= ' ' . $name . ' ' . _ORM_OP_IN 
    363. 

  363.                                         . ' (' . implode(',', $value) . ')';
    364. 

  364.                         } else {
    365. 

  365.                             throwException('字段值不合法:必须为数组', 4053);
    366. 

  366.                         }
    367. 

  367.                         break;
    368. 

  368.                     default :
    369. 

  369.                         $sql .= $name . ' ' . $where['opt'] . ' ' . $value;
    370. 

  370.                         break;
    371. 

  371.                 }
    372.
