  230.  
    231. 

  231.         if (!is_file($src_path))
    232. 

  232.             return(json_encode(array('result'=>false,"message"=>"对不起, 请刷新页面后重新上传图片")));

  233.

  234.         $src_width = 0; //原始图片的宽度
